showering
(epi-schema ((?x new_schema.v) ** ?e)
	(:Roles
		(!R1 (?A PERSON.N))
		(!R2 (?B MAN.N))
		(!R3 (?A BODY_PART.N))
		(!R4 (?E GIRL.N))
		(!R5 (?F PERSON.N))
		(!R6 (?I LIQUID.N))
		(!R7 (?J COLD.N))
		(!R8 (?K SOAP.N))
		(!R9 (?L DESTINATION.N))
		(!R10 (?M ENTITY.N))
		(!R11 (?N HAIR.N))
	)

	(:Steps
		(?E1 (?A TAKE.V ?F))
		(?E2 (?B ((ADV-A (FROM.P ?M)) SHIVER_PROTO.V) ?L))
		(?E3 (?B TAKE.V ?A))
		(?E4 (?B WASH_PROTO.V ?N))
		(?E5 (?A SHOWER_PROTO.V ?F))
		(?E6 (?A WASH_PROTO.V ?F))
		(?E7 (?E WASH_PROTO.V ?A))
		(?E8 (?I GO.V ?J))
		(?E9 (?B HAVE_PROTO.V ?K))
	)

	(:Goals
		(?G1 (?B (WANT.V (KA ((ADV-A (AT.P ?L)) BE.V)))))
		(?G2 (?B (WANT.V (THAT (NOT (?N DIRTY.A))))))
		(?G3 (?A (WANT.V (THAT (NOT (?F DIRTY.A))))))
		(?G4 (?A (WANT.V (THAT (NOT (?F DIRTY.A))))))
		(?G5 (?E (WANT.V (THAT (NOT (?A DIRTY.A))))))
	)

	(:Preconds
		(?I1 (?B (AT.P ?M)))
		(?I2 (NOT (?B (AT.P ?L))))
		(?I3 (?N DIRTY.A))
		(?I4 (NOT (?N CLEAN.A)))
		(?I5 (?F DIRTY.A))
		(?I6 (NOT (?F CLEAN.A)))
		(?I7 (?F DIRTY.A))
		(?I8 (NOT (?F CLEAN.A)))
		(?I9 (?A DIRTY.A))
		(?I10 (NOT (?A CLEAN.A)))
	)

	(:Postconds
		(?P1 (NOT (?B (AT.P ?M))))
		(?P2 (?B (AT.P ?L)))
		(?P3 (NOT (?N DIRTY.A)))
		(?P4 (?N CLEAN.A))
		(?P5 (NOT (?F DIRTY.A)))
		(?P6 (?F CLEAN.A))
		(?P7 (NOT (?F DIRTY.A)))
		(?P8 (?F CLEAN.A))
		(?P9 (NOT (?A DIRTY.A)))
		(?P10 (?A CLEAN.A))
	)

	(:Episode-relations
		(!W1 (?G1 CAUSE.V ?E2))
		(!W2 (?I1 BEFORE ?E2))
		(!W3 (?I2 BEFORE ?E2))
		(!W4 (?P1 AFTER ?E2))
		(!W5 (?P2 AFTER ?E2))
		(!W6 (?G2 CAUSE.V ?E4))
		(!W7 (?I3 BEFORE ?E4))
		(!W8 (?I4 BEFORE ?E4))
		(!W9 (?P3 AFTER ?E4))
		(!W10 (?P4 AFTER ?E4))
		(!W11 (?G3 CAUSE.V ?E5))
		(!W12 (?I5 BEFORE ?E5))
		(!W13 (?I6 BEFORE ?E5))
		(!W14 (?P5 AFTER ?E5))
		(!W15 (?P6 AFTER ?E5))
		(!W16 (?G4 CAUSE.V ?E6))
		(!W17 (?I7 BEFORE ?E6))
		(!W18 (?I8 BEFORE ?E6))
		(!W19 (?P7 AFTER ?E6))
		(!W20 (?P8 AFTER ?E6))
		(!W21 (?G5 CAUSE.V ?E7))
		(!W22 (?I9 BEFORE ?E7))
		(!W23 (?I10 BEFORE ?E7))
		(!W24 (?P9 AFTER ?E7))
		(!W25 (?P10 AFTER ?E7))
	)
)
